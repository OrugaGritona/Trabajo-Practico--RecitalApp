package main.kotlin.payments
import java.time.*

abstract class MedioDePago {
abstract val name: String

    fun obtenerTodosTiposDePago(): List<MedioDePago> = listOf(
        MercadoPago(),
        Visa(),
        MasterCard()
    )


}
